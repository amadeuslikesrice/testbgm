let helperOpen = false;
let ignoreNextContextMenu = false;

$(document).on("contextmenu", function (e) {
  e.preventDefault();

  if (ignoreNextContextMenu) {
    ignoreNextContextMenu = false;
    return;
  }

  if (helperOpen) return;

  helperOpen = true;

  $(".helper")
    .stop(true, true)
    .fadeIn(200)
    .addClass("show");
});

$(document).on("mousedown", function (e) {
  if (!helperOpen) return;

  if (e.button === 2) {
    ignoreNextContextMenu = true;
  }

  helperOpen = false;

  $(".helper")
    .stop(true, true)
    .fadeOut(200)
    .removeClass("show");
});
